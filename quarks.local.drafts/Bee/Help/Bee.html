//:Distributing patterns in a grid, off-centered


Bee.draw1({ | b, i |

var grid, pos, length, hex_length;

grid = [({ | i | (1..4).rotate(i) } ! 3).flat, ({ (1..4) } ! 3).flat].flop * 115 - 30;

while { (pos = grid.pop).notNil } {

length = 2 rrand: 6;

hex_length = 1.0 rrand: 2.0;

0.1.wait;

b.jump((pos[0])@(pos[1]));

b.orientation = 2pi.rand;

[

{ { | i | b.turn(1 / (i % 20 + 1) * pi * 1.001); b.move(length) }.repeat(100, 0.01); },

{ { | i | b.turn(pi / 3); b.move(i % 20 * hex_length) }.repeat(75, 0.01); }

].choose.value; 

}

});
