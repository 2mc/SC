// basic metagon

(

t = Turtle.draw({ | t |

var metagonSide = 64;

Pseq({ | i |

Pfuncn({ t.turn(2 * pi / (i + 3)); t.move(metagonSide, 1) }, i + 2)

} ! 50

, 1) 

},  x: 400, y: 400,rate: 0);

)


// Strange ???metagon???



(

t = Turtle.draw({ | t |

Pseq({ | i |

Pfuncn({ t.turn(2 ** (i ** 1.1) * pi / (i + 1 ** 1.1)); t.move(metagonSide, 1) }, i + 2)

} ! 50

, 1) 

},  x: 400, y: 400,rate: 0);

)
