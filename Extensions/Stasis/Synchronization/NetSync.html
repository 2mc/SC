/*

NetSync: Synchronize processes running on different computers in a network via OSC. 

*/


// SyncSender broadcasts OSC to one or more hosts


a = SyncSender(NetAddr.localAddr, Pbind(\syncEvent, '/test'));

a.start;


// SyncResponder receives OSC and broadcasts it via 'update' message to any of its dependants. 


b = SyncResponder.new;


b.addDependant({ | ... args | args.postln; });


a.clock.tempo = 2;


c = SimpleController(b, '/test', { | receiver, message, time |

format("receiver: %, message: %, beat number: %", receiver, message, time).postln

});


e = SyncAction('/test', { | receiver, message, time |

format("SYNC ACTION!!! receiver: %, message: %, beat number: %", receiver, message, time).postln

});


e.model

e.inspect;


c.remove;


b.deactivate;

b.activate;
