(

// Alternative of Archytas_Stoichos generation by help of Systema


(

// Τετρακτύς

~mese = 1/|1 ;

~octave = 2/|1 ; 

~fifth = 3/|2 ;

~forth = 4/|3 ;


~tone = ~fifth / ~forth ;

~leimma = ~forth / ~tone / ~tone ;

~leimma.info

);


// Pythagorean Tetrachord

(

~pythTetra = [ 

~mese, 

~mese / ~tone, 

~mese / (~tone**2), 

~mese / (~tone**2) / ~leimma 

];


~pythTetra.do{|r| ("\t" + r.info + "\n").postln}; ""; 

);


// Lyra of Hermes // συναφή

(

~pythSynaphe = (~forth * ~pythTetra) ++ ~pythTetra;


~pythSynaphe.do{|r| ("\t" + r.info + "\n").postln}; ""; 

);


// ἁρμονία // διάζευξις

(

~pythHarmonia = (~fifth * ~pythTetra) ++ ~pythTetra;


~pythHarmonia.do{|r| ("\t" + r.info + "\n").postln}; ""; 

);

~pythHarmonia.first / ~pythHarmonia.last == ~octave; // test

~pythHarmonia.first * ~pythHarmonia.last == ~tone;   // test


// ἁρμονία -- cyclic

(

~cyclicHarmonia = Systema(~pythHarmonia, 0, name: \cyclicHarmonia).isCyclic_(true);


~cyclicHarmonia.postld.play;

);

/*

// τρόπος -- cyclic

(

~troposRatios = (-7..7).collect{|degree| ~cyclicHarmonia.degreeToRatio(degree)}; // 7up | 7down


~troposRatios.do{|r| ("\t" + r.info + "").postln}; ""; //see them


~cyclicTropos = Systema(~troposRatios, 0, name: \cyclicTropos).isCyclic_(true);


~cyclicTropos.postld.play;

);

// three column ἁρμονία or ! τρόπος ! of Archytas' tuning

(

~root = ~pythHarmonia; // take it straight

~turnedRoot = ~root * (28/|27); // turn it by 28/27

~flippedRoot = ~root * (16/|15); // flip it by 16/15


~trinityRatios = ~flippedRoot.collect{|r, i| [r] ++ ~turnedRoot[i] ++ ~root[i] }; //collect it

~trinityRatios.do{|l| ("\t" + l.asCompileString + "\n").postln}; ""; //post it


~trinityRatios = ~trinityRatios.flat.drop(2); // drop 2 transgressing the harmonia 

~trinityRatios.do{|r| ("\t" + r.info + "").postln}; ""; //post it


~archytasTrinity = Systema(~trinityRatios, 0, name: \archytasTrinity).isCyclic_(true);


~archytasTrinity.postld.play; // play three column harmonia

~archytasTrinity.postld.play(0.3,( (-7*3) .. (7*3) ) ); // play three column tropos


~trinityTroposRatios = ((-7*3)..(7*3)).collect{|degree| ~archytasTrinity.degreeToRatio(degree)};

~trinityTropos = Systema(~trinityTroposRatios, 0, name: \trinityTropos).isCyclic_(true);


~trinityTropos.postld.play; // no 'sharps' overshoot the SYMETRICALLY created double harmonia 

// but isn't this the strongest argument for the hypothesis that it was among Pythagoreans

);


// universal Harmonia -- universal Tropos

(

~uniN = 70; ~uniN == (23 * 3 + 1); // quantity of all AGM notation signs

~naturals = 23 + 1; // number of steps of the 'natural' signs

~meseIndex = 31; // index of root/mese in Pöhlmann/West 1992 table


~stepsAboveMese = (~uniN - ~meseIndex); // steps above mese

~stepsBelowMese = ~meseIndex - 1; // steps below mese

~uniDegrees = (~stepsAboveMese.neg .. ~stepsBelowMese); // scale degrees of all signs

~uniDegrees.size == ~uniN;


~archytasTrinity.play(0.2, ~uniDegrees); // play them right away, or sample the tuning:


~uniRatios3harmoniai = ~uniDegrees.collect{|degree| ~archytasTrinity.degreeToRatio(degree)};

~uniRatios3tropoi = ~uniDegrees.collect{|degree| ~trinityTropos.degreeToRatio(degree)};

~uniRatios3harmoniai == ~uniRatios3tropoi;

// creation of universal Harmonia -- universal Tropos from 'naturals'

(

~naturalsAboveMese = (~uniN - ~meseIndex) div: 3; // 'naturals' above mese

~naturalsBelowMese = ~meseIndex div: 3 ; // 'naturals' below mese

~naturalsDegrees = (~naturalsAboveMese.neg .. ~naturalsBelowMese); //scale degrees of 'naturals'

~naturalsDegrees.size == (23 + 1); // 24 'naturals'


~uniHarmoniaRatios = ~naturalsDegrees.collect{|degree| ~cyclicHarmonia.degreeToRatio(degree)};

~uniTroposRatios = ~naturalsDegrees.collect{|degree| ~cyclicTropos.degreeToRatio(degree)};

~uniHarmoniaRatios == ~uniTroposRatios;

~uniHarmonia.postld.play;

);


// universal column trinity

(

~base = ~uniHarmoniaRatios; // take it straight

~turned = ~uniHarmoniaRatios * (28/|27); // turn it by 28/27

~flipped = ~uniHarmoniaRatios * (16/|15); // flip it by 16/15


~uniTritiyRatios = ~flipped.collect{|r, i| [r] ++ ~turned[i] ++ ~base[i] }; //collect it

~uniTritiyRatios.do{|l| ("\t" + l.asCompileString + "\n").postln}; ""; //post it


~uniTritiyRatios = ~uniTritiyRatios.flat.drop(2); // drop 2 non attested 'sharps' on top 

~uniTritiyRatios.do{|r| ("\t" + r.info + "").postln}; ""; //post it


~uniTritiyRatios.size == ~uniN; // 70 ratios for 70 signs

~uniTritiyRatios == ~uniRatios3harmoniai; // equivalence of generation

~uniTritiyRatios == ~uniRatios3tropoi ; // equivalence of generation

~uniTrinity.postlid;
